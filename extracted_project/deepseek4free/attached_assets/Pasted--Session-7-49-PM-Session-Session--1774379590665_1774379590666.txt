Session (الجلسة)
Session هي مجموعة من البيانات التي تحفظ حالة المستخدم أثناء تصفحه للموقع.

تتكون من:
1. Cookies 🍪
ملفات صغيرة يحفظها المتصفح تحتوي على:

Session ID
Token
تفضيلات المستخدم
2. LocalStorage 💾
مساحة تخزين في المتصفح تحتوي على:

بيانات التطبيق
إعدادات
معلومات مؤقتة
3. SessionStorage ⏱️
مثل localStorage لكن:

تُمسح عند إغلاق التبويب
مؤقتة أكثر
الفرق بين Token و Session:
Token	Session
مفتاح واحد فقط	Cookies + Storage
ينتهي أسرع	يدوم أطول
يحتاج تجديد متكرر	نادراً يحتاج تجديد
مثال عملي:
عندما تسجل دخول في DeepSeek:

الموقع يعطيك Token (مفتاح)
يحفظ Cookies (معلومات الجلسة)
يحفظ localStorage (بيانات التطبيق)
Session = Token + Cookies + Storage

باختصار: Session = كل شيء يجعلك "مسجل دخول" دون الحاجة لإعادة تسجيل الدخول.

"""
DeepSeek Session Manager
إدارة Session كامل بدلاً من Token فقط
"""
import json
import pickle
from pathlib import Path
from typing import Dict, Optional

class SessionManager:
    def __init__(self, session_file='deepseek_session.pkl'):
        self.session_file = Path(session_file)
        self.session_data = self._load_session()
    
    def _load_session(self) -> Optional[Dict]:
        """تحميل Session من الملف"""
        if not self.session_file.exists():
            return None
        
        try:
            with open(self.session_file, 'rb') as f:
                return pickle.load(f)
        except Exception:
            return None
    
    def save_session(self, cookies: Dict, local_storage: Dict, token: str):
        """حفظ Session كامل"""
        session_data = {
            'cookies': cookies,
            'local_storage': local_storage,
            'token': token
        }
        
        with open(self.session_file, 'wb') as f:
            pickle.dump(session_data, f)
        
        self.session_data = session_data
    
    def get_token(self) -> Optional[str]:
        """استخراج Token من Session"""
        if not self.session_data:
            return None
        return self.session_data.get('token')
    
    def get_cookies(self) -> Dict:
        """استخراج Cookies من Session"""
        if not self.session_data:
            return {}
        return self.session_data.get('cookies', {})
    
    def is_valid(self) -> bool:
        """التحقق من صلاحية Session"""
        return self.session_data is not None and 'token' in self.session_data

# Route لصفحة الإعداد
@app.route("/setup")
def setup():
    return render_template("setup.html")

# Route لحفظ Session
@app.route("/api/save-session", methods=["POST"])
def save_session():
    try:
        data = request.json
        
        # استخراج البيانات
        cookies_str = data.get('cookies', '')
        local_storage = data.get('localStorage', {})
        token = data.get('token', '')
        
        if not token:
            return {"success": False, "error": "Token not found"}, 400
        
        # تحويل cookies من string إلى dict
        cookies_dict = {}
        for cookie in cookies_str.split('; '):
            if '=' in cookie:
                key, value = cookie.split('=', 1)
                cookies_dict[key] = value
        
        # حفظ Session
        session_manager.save_session(
            cookies=cookies_dict,
            local_storage=local_storage,
            token=token
        )
        
        # إعادة تحميل API
        global _api
        _api = None
        
        return {"success": True, "message": "Session saved successfully"}
        
    except Exception as e:
        return {"success": False, "error": str(e)}, 500
